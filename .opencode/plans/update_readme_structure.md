```bash
# Проверить, что все ссылки ведут на существующие файлы
find . -name "*.md" -type f | xargs grep -l "\[.*\]\(.*\)" | while read file; do
  echo "Проверка ссылок в $file:"
  grep -o "\[.*\]\([^)]*\)" "$file" | while read link; do
    target=$(echo "$link" | sed -n 's/.*(\([^)]*\)).*/\1/p')
    if [ ! -f "$target" ] && [ ! -d "$target" ]; then
      echo "  НЕ НАЙДЕНО: $target"
    fi
  done
done
```
